export default {
  data() {

    return {
      form: {
        username: "",
        password: ""
      }
    }

  },
  methods:{
    login(form){
      this.$refs[form].validate((valid)=>{
        if(valid){
          console.log("验证通过")

              res=>{
                console.log(res)
                if(res.data.status === 200){
                  localStorage.setItem("username",res.data.username)
                  this.$message({
                    message:res.data.message,
                    type:'success'
                  })
                  this.$router.push('/home')
                }
              }
          )
              .catch(err=>{
                console.error(err)
              })
        }else {
          console.error("验证不通过")
        }
      })
    }
  }
}
</script>
