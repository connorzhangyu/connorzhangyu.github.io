#Centos8  #VirtualBox 

VirtualBox增强功能Centos8⌗


# 需要安装一定的依赖，例如：make gcc perl
sudo yum install gcc perl make tar bzip2 elfutils-libelf-devel

#同时安装的kernel-devel版本要和内核版本一致，安装对应版本
sudo yum install -y "kernel-devel-uname-r == $(uname -r)"

# 或者通过DNF安装所有：
dnf install tar bzip2 kernel-devel-$(uname -r) kernel-headers perl gcc make elfutils-libelf-devel
